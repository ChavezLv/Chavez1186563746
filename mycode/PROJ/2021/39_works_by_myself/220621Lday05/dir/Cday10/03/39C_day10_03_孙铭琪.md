- (a)

```c
// 这里写a的代码或者答案
int binarySearch(int *arr, int low, int high, int target) {
	int mid;
	while (mid <= high) {
		mid = (low + high) / 2;
		if (arr[mid] < target) {
			low = mid + 1;
		}
		else if (arr[mid] > target) {
			high = mid - 1;
		}
		else {
			return mid;
		}
	}
	return -1;
}
```
